CREATE TABLE phieu_nguyen_vong_nhom (
    NhomID BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    MonHocID INT UNSIGNED NOT NULL, 
    HocKyID INT UNSIGNED NOT NULL, 
    Thu INT NOT NULL, 
    BuoiHoc ENUM('Sáng', 'Chiều', 'Tối') NOT NULL,
    SoLuongHienTai INT DEFAULT 0, 
    TrangThai ENUM('Đang gom', 'Đã mở lớp', 'Hủy bỏ') DEFAULT 'Đang gom',
    created_at TIMESTAMP NULL,
    updated_at TIMESTAMP NULL,
    -- Vẫn giữ Unique để tránh một môn có 2 nhóm cùng lịch trong 1 kỳ
    UNIQUE KEY uk_mon_hk_lich (MonHocID, HocKyID, Thu, BuoiHoc)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE chi_tiet_nguyen_vong (
    ChiTietID BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    NhomID BIGINT UNSIGNED NOT NULL,
    SinhVienID BIGINT UNSIGNED NOT NULL, 
    NgayDangKy TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_sv_nhom_don (NhomID, SinhVienID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

DROP VIEW IF EXISTS v_lich_hoc_sinhvien;

CREATE VIEW v_lich_hoc_sinhvien AS
SELECT 
    dk.SinhVienID,
    sv.MaSV,
    sv.HoTen,
    lh.LichHocID,
    lh.LopHocPhanID,
    lh.NgayHoc,
    lh.Thu, -- Cột mới thêm vào
    lh.BuoiHoc,
    lh.TietBatDau,
    lh.SoTiet,
    lh.PhongHoc,
    lh.GhiChu,
    mh.TenMon,
    lhp.MaLopHP,
    gv.HoTen AS TenGiangVien,
    lhp.HocKyID,
    hk.TenHocKy,
    nh.NamHocID,
    nh.TenNamHoc,
    lhp.NgayBatDau AS NgayBatDauLop,
    lhp.NgayKetThuc AS NgayKetThucLop,
    hk.NgayBatDau AS NgayBatDauHocKy,
    hk.NgayKetThuc AS NgayKetThucHocKy
FROM lichhoc lh
JOIN lophocphan lhp ON lh.LopHocPhanID = lhp.LopHocPhanID
JOIN monhoc mh ON lhp.MonHocID = mh.MonHocID
JOIN hocky hk ON lhp.HocKyID = hk.HocKyID
JOIN namhoc nh ON hk.NamHocID = nh.NamHocID
JOIN dangkyhocphan dk ON lhp.LopHocPhanID = dk.LopHocPhanID
JOIN sinhvien sv ON dk.SinhVienID = sv.SinhVienID
LEFT JOIN giangvien gv ON lhp.GiangVienID = gv.GiangVienID;
